ABI ... */]; async function makePayment() { const provider = new ethers.providers.Web3Provider(window.ethereum); await provider.send("eth_requestAccounts", []); const signer = provider.getSigner(); const contract = new ethers.Contract(contractAddress, abi, signer); const orderId = document.getElementById("orderId").value; const amount = ethers.utils.parseEther(document.getElementById("amount").value); try { const tx = await contract.pay(orderId, { value: amount }); await tx.wait(); alert("Payment successful!"); } catch (err) { alert("Payment failed: " + err.message); } }
